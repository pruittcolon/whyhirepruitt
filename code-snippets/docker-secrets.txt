# Docker Secrets Management
# k8s/base/secrets-management.yaml
#
# Secure secrets handling:
# - External secret loading
# - Volume-mounted secrets
# - Secret rotation
# - Access control

apiVersion: v1
kind: Secret
metadata:
  name: app-secrets
  namespace: nemo-system
type: Opaque
stringData:
  jwt-secret-key: "${JWT_SECRET_KEY}"
  db-encryption-key: "${DB_ENCRYPTION_KEY}"
  redis-password: "${REDIS_PASSWORD}"

---
# External Secrets Operator integration
apiVersion: external-secrets.io/v1beta1
kind: ExternalSecret
metadata:
  name: vault-secrets
  namespace: nemo-system
spec:
  refreshInterval: 1h
  secretStoreRef:
    name: vault-backend
    kind: SecretStore
  target:
    name: production-secrets
    creationPolicy: Owner
  data:
    - secretKey: jwt-signing-key
      remoteRef:
        key: nemo/production/jwt
        property: signing_key
    - secretKey: database-key
      remoteRef:
        key: nemo/production/encryption
        property: db_key

---
# Secret volume mount example
apiVersion: apps/v1
kind: Deployment
metadata:
  name: secure-service
spec:
  template:
    spec:
      containers:
        - name: app
          image: nemo/api-gateway:latest
          env:
            - name: JWT_SECRET
              valueFrom:
                secretKeyRef:
                  name: app-secrets
                  key: jwt-secret-key
          volumeMounts:
            - name: secrets
              mountPath: /run/secrets
              readOnly: true
      volumes:
        - name: secrets
          secret:
            secretName: app-secrets
            items:
              - key: db-encryption-key
                path: db_encryption_key
                mode: 0400

---
# RBAC for secret access
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: secret-reader
  namespace: nemo-system
rules:
  - apiGroups: [""]
    resources: ["secrets"]
    resourceNames: ["app-secrets"]
    verbs: ["get"]

---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: read-secrets
  namespace: nemo-system
subjects:
  - kind: ServiceAccount
    name: nemo-service-account
roleRef:
  kind: Role
  name: secret-reader
  apiGroup: rbac.authorization.k8s.io
